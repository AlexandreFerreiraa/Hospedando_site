library(plm)

GERANDO OS DADOS (Salrio vs Educao)

N <- 100       # Indivduos
T_per <- 30     # Perodos
BETA_REAL <- 0.40 # Retorno real da educao (40%)

# Habilidade no observada (Inteligencia)
alpha_i <- rep(rnorm(N), each = T_per)

# Educao (X) correlacionada com Habilidade (Endogeneidade)
X_educ <- 12 + (2 * alpha_i) + rnorm(N*T_per, sd = 2)

# Salrio (Y) depende de Educao e Habilidade
Y_wage <- (BETA_REAL * X_educ) + (0.5 * alpha_i) + rnorm(N*T_per, sd = 2)

dados <- data.frame(id=factor(rep(1:N, each=T_per)), time=factor(rep(1:T_per, N)), Y=Y_wage, X=X_educ)

ESTIMADOR DE EFEITOS FIXOS

dados_transformed <- dados %>%
  group_by(id) %>%
  mutate(
    Y_demeaned = Y - mean(Y),  
    X_demeaned = X - mean(X)  
  ) %>%
  ungroup()

modelo_fe_manual <- lm(Y_demeaned ~ X_demeaned - 1, data = dados_transformed)

b_fe_manual <- coef(modelo_fe_manual)["X_demeaned"]

ESTIMADOR DE EFEITOS ALEATRIOS

p_dados <- pdata.frame(dados, index=c("id", "time"))

# Precisamos rodar o PLM para pegar as matrizes de varincia corretas
mod_fe_plm <- plm(Y ~ X, data = p_dados, model = "within") # S para pegar a Var-Cov correta
mod_re_plm <- plm(Y ~ X, data = p_dados, model = "random")

# Betas e Matrizes de Covarincia
b_fe <- coef(mod_fe_plm)["X"]
b_re <- coef(mod_re_plm)["X"]
V_fe <- vcov(mod_fe_plm)[1,1] # Varincia do beta FE
V_re <- vcov(mod_re_plm)[2,2] # Varincia do beta RE (ndice 2 pois o 1  intercepto)

TESTE DE HAUSMAN

# Frmula: H = (b_fe - b_re)' * [Var(b_fe) - Var(b_re)]^-1 * (b_fe - b_re)

# Passo A: Vetor de Diferenas (q)
q_hat <- b_fe - b_re

# Passo B: Diferena das Varincias (Baseado no Lema de Hausman)
# Aqui aplicamos a propriedade de eficincia: Var(Diff) = Var(FE) - Var(RE)
Var_diff <- V_fe - V_re

# Passo C: Estatstica H (Chi-Quadrado)
# Como temos apenas 1 regressor, a inversa da matriz  apenas 1/Var
H_stat_manual <- (q_hat^2) / Var_diff
H_stat_manual <- as.numeric(H_stat_manual)

# Passo D: P-valor
p_val_manual <- pchisq(H_stat_manual, df = 1, lower.tail = FALSE)

TESTE AUTOMTICO (para comparao)

teste_automatico <- phtest(mod_fe_plm, mod_re_plm)

int_fe <- mean(dados$Y) - (b_fe * mean(dados$X))
int_re <- coef(mod_re_plm)[1]
